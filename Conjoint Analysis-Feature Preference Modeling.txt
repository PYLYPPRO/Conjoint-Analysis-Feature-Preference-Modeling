<script>
// --- Conjoint Analysis Simulation in JavaScript ---

// This simulates the derived Part-Worth Utilities (PWU) for each attribute level, 
// based on previous market research data. Higher PWU means higher preference.
const partWorthUtilities = {
    cost: {
        '10': -5,  // High Cost: Negative Utility
        '20': 5,   // Medium Cost: Positive Utility
        '30': 15   // Low Cost: High Positive Utility
    },
    format: {
        '15': 2,   // On-Campus: Low positive utility
        '25': 8,   // Hybrid: Medium utility
        '35': 12   // Fully Online: High utility (for target segment)
    },
    career_services: {
        '5': 0,   // Standard: Baseline utility
        '15': 10  // Dedicated Internship: High utility
    },
    course_scheduling: {
        '10': 3,   // Fixed: Low utility
        '20': 7    // Flexible: Higher utility
    }
};

function runConjointAnalysis() {
    // 1. Collect user choices (representing a potential program profile)
    const costLevel = document.getElementById('cost').value;
    const formatLevel = document.getElementById('format').value;
    const careerServicesLevel = document.getElementById('career_services').value;
    const schedulingLevel = document.getElementById('course_scheduling').value;
    const satisfaction = parseInt(document.getElementById('satisfaction_rating').value);

    // 2. Calculate the Total Program Utility Score
    const utilityCost = partWorthUtilities.cost[costLevel];
    const utilityFormat = partWorthUtilities.format[formatLevel];
    const utilityCareer = partWorthUtilities.career_services[careerServicesLevel];
    const utilityScheduling = partWorthUtilities.course_scheduling[schedulingLevel];

    const totalUtilityScore = utilityCost + utilityFormat + utilityCareer + utilityScheduling;

    // 3. Estimate Market Share (Simulated) based on utility and competition (using a Logit model concept)
    // We assume a baseline competitor has a utility of 20.
    const competitionUtility = 20;
    const numerator = Math.exp(totalUtilityScore * 0.1);
    const denominator = numerator + Math.exp(competitionUtility * 0.1);
    const simulatedMarketShare = (numerator / denominator) * 100;

    // 4. Calculate Customer Lifetime Value (CLV) correlation (Simulated)
    // CLV is often correlated with satisfaction and low cost sensitivity
    const clvIndex = totalUtilityScore * 0.8 + satisfaction * 2;
    
    // 5. Output and Detailed Interpretation
    const resultDiv = document.getElementById('result_conjoint');

    let message = `<h2>CONJOINT ANALYSIS REPORT</h2>`;
    message += `<h3>1. PROGRAM PROFILE UTILITY CALCULATION</h3>`;
    message += `<p><strong>Total Program Utility Score: ${totalUtilityScore}</strong> (Baseline Competitor Utility: ${competitionUtility})</p>`;
    message += `<ul>
        <li>Cost Utility: ${utilityCost}</li>
        <li>Format Utility: ${utilityFormat}</li>
        <li>Career Services Utility: ${utilityCareer}</li>
        <li>Scheduling Utility: ${utilityScheduling}</li>
    </ul>`;

    message += `<h3>2. MARKETING INSIGHTS</h3>`;
    message += `<p><strong>Simulated Market Share (vs. Competitor):</strong> ${simulatedMarketShare.toFixed(2)}%</p>`;
    message += `<p><strong>Estimated CLV Index (Higher is Better):</strong> ${clvIndex.toFixed(1)}</p>`;

    if (totalUtilityScore > 35) {
        message += `<p><strong>CONCLUSION: High Program Attractiveness.</strong> This combination of features (especially the **Low Cost** and **Fully Online** format) demonstrates maximum appeal for the target segment. The program is optimized for both market share and CLV.</p>`;
    } else if (totalUtilityScore < 10) {
        message += `<p><strong>CONCLUSION: Low Program Attractiveness.</strong> The combination (likely high cost and fixed schedule) results in low utility. **Recommendation:** Rework the pricing structure and consider offering more flexible delivery formats to increase competitive advantage.</p>`;
    } else {
        message += `<p><strong>CONCLUSION: Medium Attractiveness.</strong> The program is viable but lacks a strong differentiator. **Recommendation:** Focus marketing efforts on the high-utility features selected (e.g., Dedicated Internship Placement) to maximize enrollment.</p>`;
    }
    
    resultDiv.innerHTML = message;
}
</script>
